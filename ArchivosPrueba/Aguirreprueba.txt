bandera=true
while(bandera==true)
  puts "***Bienvenido***** "
  puts "Selecionar las siguientes opciones "
  puts "1. Pedir edad "
  puts "2.  Saludar "
  puts "3. Salir  "
  respuesta=gets.chomp
  case respuesta
  when "1"
     puts "Dime tu edad: "
     edad=gets.chomp
    if edad<18
      puts "Eres menor de edad"
    else
       puts "Eres mayor de edad"
  when "2"
      puts "Dime tu nombre"
      nombre=gets.chomp
      puts "Hola, #{nombre}"
   when "3"
     bandera=false

  end
end
